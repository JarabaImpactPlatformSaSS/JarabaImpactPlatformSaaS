export function calculateMaturityScore(answers) {
  const { presence, operations, ai } = answers;
  const { weights: w, maxScores: max } = CONFIG;
  
  const normalized = 
    (presence / max.presence) * w.presence +
    (operations / max.operations) * w.operations +
    (ai / max.ai) * w.ai;
  
  return Math.round(normalized * 10);
}

export function getLevel(score) {
  for (const [key, level] of Object.entries(CONFIG.levels)) {
    if (score >= level.min && score <= level.max) {
      return { key, ...level };
    }
  }
  return CONFIG.levels.analog;
}

export function calculateEconomicImpact(answers) {
  const { presenceFlag, volumeFlag, operationsFlag } = answers;
  
  const base = CONFIG.volumeBases[volumeFlag];
  const multiplier = CONFIG.presenceMultipliers[presenceFlag];
  const hoursLost = CONFIG.hoursLost[operationsFlag];
  
  const monthlyLost = Math.round(base.monthly * multiplier);
  
  return {
    yearlyLost: monthlyLost * 12,
    monthlyLost,
    potentialSearches: Math.round(base.searches * multiplier),
    hoursLostWeekly: hoursLost,
    hoursLostYearly: hoursLost * 52
  };
}

export function identifyPrimaryGap(answers, score) {
  const { presence, operations, ai } = answers;
  
  if (presence === 0) return 'presence_none';
  if (presence === 1) return 'presence_social_only';
  if (operations === 0) return 'ops_analog';
  if (ai === 0) return 'ai_unaware';
  if (operations === 1) return 'ops_spreadsheet';
  if (ai === 1) return 'ai_aware_not_using';
  if (score >= 7) return 'optimization';
  
  return 'presence_none';
}

export function generateFullResult(answers) {
  const score = calculateMaturityScore(answers);
  const level = getLevel(score);
  const economic = calculateEconomicImpact(answers);
  const gapKey = identifyPrimaryGap(answers, score);
  const gap = CONFIG.gaps[gapKey];
  
  return {
    score,
    level,
    economic,
    gap: { key: gapKey, ...gap },
    answers,
    timestamp: new Date().toISOString()
  };
}
